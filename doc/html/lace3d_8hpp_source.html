393  template<class VALTYPE>
394  vector3d<typename lace::numeric_type<VALTYPE>::complex> solve_cubeq(VALTYPE a, VALTYPE b, VALTYPE c, VALTYPE d)
395  // Solves ax^3 + bx^2 + cx + d = 0
396  {
397  b /= a;
398  c /= a;
399  d /= a;
400  typename lace::numeric_type<VALTYPE>::complex I(VALTYPE(0),VALTYPE(1));
401  VALTYPE aa = c/3 - b*b/9, bb = d/2 - b*c/6 + b*b*b/27;
402  VALTYPE discr1 = aa*aa*aa + bb*bb;
403  typename lace::numeric_type<VALTYPE>::complex discr2;
404  if (discr1 > VALTYPE(0))
405  {
406  VALTYPE discr23 = std::sqrt(discr1) - bb;
407  int sgn = (discr23>0)? 1 : -1;
408  discr23 *= sgn;
409  discr2 = sgn*std::exp(std::log(discr23)/3);
410  }
411  else
412  {
413  discr2 = -bb + std::sqrt(-discr1)*I;
414  VALTYPE dabs = lace::abs(discr2), darg = lace::arg(discr2);
415  discr2 = lace::polar(std::exp(std::log(dabs)/3),darg/3);
416  }
417  typename lace::numeric_type<VALTYPE>::complex r1 = discr2 - aa/discr2, r2 = discr2 + aa/discr2;
418  return vector3d<typename lace::numeric_type<VALTYPE>::complex>( r1 - b/3,
419  -r1/2 + std::sqrt(3.0)*r2*I/2 - b/3,
420  -r1/2 - std::sqrt(3.0)*r2*I/2 - b/3);
421  }
